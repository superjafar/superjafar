class introduction:
      def __init__(self, nama, umur, status):
          self.nama   = nama
          self.umur   = umur
          self.status = status

      def info(self):
          print("Nama : ", self.nama)
          print("Umur : ", self.umur)
          print("Asal : ", self.status)

jafar = introduction("Jafar", 19, "Mahasiswa")

jafar.info()
